022package org.biojava.utils.lsid;
023
024/**
025 * Exception thrown in the event of an error in
026 * parsing a LSID-formatted string.
027 *
028 * @author Michael Heuer
029 */
030public class LifeScienceIdentifierParseException
031    extends Exception {
032
033    /**
034     * Construct a new parse exception with no message.
035     */
036    public LifeScienceIdentifierParseException() {
037        super();
038    }
039
040    /**
041     * Construct a new parse exception with the
042     * specified error message.
043     *
044     * @param message error message
045     */
046    public LifeScienceIdentifierParseException(String message) {
047        super(message);
048    }
049}
