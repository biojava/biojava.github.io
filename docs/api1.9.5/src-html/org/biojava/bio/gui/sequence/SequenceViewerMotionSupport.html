021package org.biojava.bio.gui.sequence;
022
023import java.util.ArrayList;
024import java.util.Iterator;
025import java.util.List;
026
027/**
028 * @author Matthew Pocock
029 */
030public class SequenceViewerMotionSupport {
031  private List listeners = new ArrayList();
032  
033  public void addSequenceViewerMotionListener(SequenceViewerMotionListener svl) {
034    synchronized(listeners) {
035      listeners.add(svl);
036    }
037  }
038  
039  public void removeSequenceViewerMotionListener(SequenceViewerMotionListener svl) {
040    synchronized(listeners) {
041      listeners.remove(svl);
042    }
043  }
044  
045  public void fireMouseDragged(SequenceViewerEvent sve) {
046    List l;
047    synchronized(listeners) {
048      l = new ArrayList(listeners);
049    }
050    for(Iterator i = l.iterator(); i.hasNext(); ) {
051      SequenceViewerMotionListener svml = (SequenceViewerMotionListener) i.next();
052      svml.mouseDragged(sve);
053    }
054  }
055  
056  public void fireMouseMoved(SequenceViewerEvent sve) {
057    List l;
058    synchronized(listeners) {
059      l = new ArrayList(listeners);
060    }
061    for(Iterator i = l.iterator(); i.hasNext(); ) {
062      SequenceViewerMotionListener svml = (SequenceViewerMotionListener) i.next();
063      svml.mouseMoved(sve);
064    }
065  }
066}
