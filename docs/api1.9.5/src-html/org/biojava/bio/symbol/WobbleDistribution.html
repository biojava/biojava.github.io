022package org.biojava.bio.symbol;
023
024import java.util.Set;
025
026import org.biojava.bio.dist.Distribution;
027
028/**
029 * an object to return statistics about
030 * the frequency of the wobble base
031 * in a set of synonymous codons.
032 *
033 * @author David Huen
034 * @since 1.3
035 */
036public interface WobbleDistribution
037{
038    /**
039     * returns the residue encoded by this WobbleDistribution
040     */
041    public Symbol getResidue();
042
043    /**
044     * returns Set containing the nonWobbleBases that
045     * occur in codons that encode this residue
046     */
047    public Set getNonWobbleBases();
048
049    /**
050     * returns the frequency with which
051     * synonymous codons start with a
052     * specified pair of bases.
053     */
054    public Distribution getFrequencyOfNonWobbleBases();
055
056    /**
057     * returns the frequency of a specific
058     * wobble base in a set of synonymous
059     * codons that start with the same two
060     * bases. (sums to one over this set).
061     */
062    public Distribution getWobbleFrequency(Symbol nonWobbleBases);
063}
