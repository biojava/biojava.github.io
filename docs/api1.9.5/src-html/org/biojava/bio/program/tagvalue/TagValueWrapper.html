022package org.biojava.bio.program.tagvalue;
023
024
025/**
026 * <p>
027 * Interface for TagValueListeners that wrap other TagValueListeners
028 * </p>
029 *
030 * <p>
031 * Implementations will tend to intercept the tags or values as they stream
032 * through and modify them in some manner before forwarding them to the delegate
033 * listener. Using classes derived from SimpleTagValueWrapper, it is possible to build
034 * up complex chains of handlers that process and collate information as it
035 * streams through.
036 * </p>
037 *
038 * @author Matthew Pocock
039 * @author David Huen (conversion to interface)
040 * @since 1.2
041 */
042public interface TagValueWrapper
043  extends
044    TagValueListener
045{
046  /**
047   * get listener to which all calls will be delegated
048   */  
049  public TagValueListener getDelegate();
050
051  /**
052   * set listener to which all calls will be delegated
053   */
054  public void setDelegate(TagValueListener delegate);
055}
