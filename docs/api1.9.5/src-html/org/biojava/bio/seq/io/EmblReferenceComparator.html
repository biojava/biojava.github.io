026package org.biojava.bio.seq.io;
027
028import java.util.ArrayList;
029import java.util.Comparator;
030import java.util.List;
031
032public class EmblReferenceComparator implements Comparator {
033
034    static final Comparator INSTANCE = new EmblReferenceComparator();
035
036    private List tagOrder;
037
038    {
039        tagOrder = new ArrayList();
040        tagOrder.add(EmblLikeFormat.REFERENCE_TAG);
041        tagOrder.add(EmblLikeFormat.COORDINATE_TAG);
042        tagOrder.add(EmblLikeFormat.REF_ACCESSION_TAG);
043        tagOrder.add(EmblLikeFormat.AUTHORS_TAG);
044        tagOrder.add(EmblLikeFormat.TITLE_TAG);
045        tagOrder.add(EmblLikeFormat.JOURNAL_TAG);
046        tagOrder.add(EmblLikeFormat.REF_XREF_TAG);
047        tagOrder.add(EmblLikeFormat.SEPARATOR_TAG);
048    }
049
050    public int compare(Object o1, Object o2)
051    {
052        int index1 = tagOrder.indexOf(o1);
053        int index2 = tagOrder.indexOf(o2);
054
055        return (index1 - index2);
056    }
057
058}
