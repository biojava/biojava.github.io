022package org.biojavax.bio.db;
023
024import org.biojava.bio.seq.SequenceIterator;
025import org.biojava.bio.seq.db.SequenceDB;
026import org.biojavax.bio.seq.RichSequenceIterator;
027
028/**
029 * A database of RichSequences with accessible keys and iterators over all
030 * sequences.
031 * <p>
032 * This may have several implementations with rich behaviour, but basically most
033 * of the time you will just use the interface methods to do stuff. A sequence
034 * database contains a finite number of sequences stored under unique keys.
035 *
036 * @author Matthew Pocock
037 * @author Gerald Loeffler
038 * @author Thomas Down
039 * @author Richard Holland
040 * @since 1.5
041 */
042public interface RichSequenceDB extends SequenceDB,BioEntryDB,RichSequenceDBLite {
043    /**
044     * {@inheritDoc}
045     * Will always return an instance of RichSequenceIterator.
046     */
047    public SequenceIterator sequenceIterator();
048    
049    /**
050     * Returns a RichSequenceIterator over all sequences in the database. The order
051     * of retrieval is undefined.
052     * @return a RichSequenceIterator over all sequences
053     */
054    public RichSequenceIterator getRichSequenceIterator();
055}
