021package org.biojava.bio.seq;
022
023/**
024 * Process strings and return strand objects.
025 *
026 * @author Matthew Pocock
027 */
028public final class StrandParser {
029  public static final StrandedFeature.Strand parseStrand(String strand) {
030    if(
031      "+".equals(strand) ||
032      "1".equals(strand) ||
033      "plus".equals(strand) ||
034      "POSITIVE".equals(strand)
035    ) {
036      return StrandedFeature.POSITIVE;
037    } else if(
038      "-".equals(strand) ||
039      "-1".equals(strand) ||
040      "minus".equals(strand) ||
041      "NEGATIVE".equals(strand)
042    ) {
043      return StrandedFeature.NEGATIVE;
044    } else if(
045      ".".equals(strand) ||
046      "0".equals(strand) ||
047      "none".equals(strand) ||
048      "UNKNOWN".equals(strand)
049    ) {
050      return StrandedFeature.UNKNOWN;
051    }
052
053    throw new IllegalArgumentException("Not a legal strand: " + strand);
054  }
055}
