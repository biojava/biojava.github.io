022package org.biojava.bio.alignment;
023
024import org.biojava.bio.BioError;
025import org.biojava.bio.BioException;
026import org.biojava.bio.symbol.Location;
027import org.biojava.bio.symbol.SymbolList;
028
029
030/** 
031 * <p>SimpleSimpleAlignment is a simple implementation of
032 * AlignmentElement.</p>
033 *
034 * @author David Waring
035 */
036
037public class SimpleAlignmentElement implements AlignmentElement{
038    
039    protected String label;
040    protected Location loc;
041    protected SymbolList seq;
042    
043    public SimpleAlignmentElement(String label,SymbolList seq,Location loc)throws BioException{
044        if ((loc.getMax() - loc.getMin() + 1) != seq.length()){
045            throw new BioException("Sequence length and location length do not match");
046        }
047        this.label = label;
048        this.seq = seq;
049        this.loc = loc;
050    }
051    
052    public String getLabel(){
053        return label;
054    }
055    public Location getLoc(){
056        return loc;
057    }
058    public SymbolList getSymbolList(){
059        return seq;
060    }
061    public void setLoc(Location nLoc)throws BioError{
062         if ((nLoc.getMax() - nLoc.getMin() + 1) != seq.length()){
063            throw new BioError("Sequence length and location length do not match");
064        }
065        this.loc = nLoc;
066    }
067
068}
