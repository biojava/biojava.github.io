023package org.biojava.nbio.core.sequence.io.template;
024
025import org.biojava.nbio.core.exceptions.CompoundNotFoundException;
026import org.biojava.nbio.core.sequence.template.AbstractSequence;
027import org.biojava.nbio.core.sequence.template.Compound;
028import org.biojava.nbio.core.sequence.template.ProxySequenceReader;
029
030import java.io.IOException;
031import java.util.List;
032
033/**
034 *
035 * @author Scooter Willis <willishf at gmail dot com>
036 */
037public interface SequenceCreatorInterface<C extends Compound> {
038        /**
039         *
040         * @param sequence
041         * @param index
042         * @return
043         * @throws CompoundNotFoundException
044         * @throws IOException
045         */
046        public AbstractSequence<C> getSequence(String sequence, long index) throws CompoundNotFoundException, IOException;
047
048        /**
049         *
050         * @param proxyLoader
051         * @param index
052         * @return
053         */
054        public AbstractSequence<C> getSequence(ProxySequenceReader<C> proxyLoader, long index);
055
056        /**
057         *
058         * @param list
059         * @return
060         */
061        public AbstractSequence<C> getSequence(List<C> list);
062
063}
